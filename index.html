√ Chia Căn (Sqrt Decomposition)

Kỹ thuật chia mảng thành các khối kích thước √n để tối ưu truy vấn và cập nhật trong C++

📊 O(√n) mỗi truy vấn ⚡ Dễ cài đặt 🎯 Linh hoạt cao

📌 Phần 1: Tổng quan

Chia căn là gì?

Chia căn (Sqrt Decomposition) là kỹ thuật chia mảng gồm n phần tử thành các khối (block), mỗi khối có kích thước khoảng √n.

Mỗi khối sẽ tiền xử lý một thông tin nào đó (tổng, min, max...) để khi truy vấn, ta không cần duyệt toàn bộ mảng.

💡 Ý tưởng chính
Thay vì duyệt từng phần tử O(n), ta chia thành √n khối → duyệt tối đa √n khối + √n phần tử lẻ = O(√n)

Khi nào dùng?

Tình huống Dùng chia căn?
Truy vấn tổng/min/max đoạn + cập nhật điểm✅ Rất phù hợp
Cập nhật đoạn + truy vấn điểm✅ Phù hợp
Bài toán offline nhiều truy vấn (Mo's Algorithm)✅ Tối ưu nhất
n ≤ 10⁵ và cần code nhanh✅ Nên dùng
n ≥ 10⁶ cần O(log n)❌ Dùng Segment Tree

📐 Công thức quan trọng

Khối k bắt đầu từ: start = k * block_size

🔨 Phần 3: Xây dựng (Build)

Thuật toán

Bắt đầu: Có mảng a[n]
↓
Tính block_size = √n
↓
Duyệt i = 0 → n-1
↓
block[i / block_size] += a[i]
↓
Hoàn thành! O(n)

Code C++

C++ Build 
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 100005;
int a[MAXN];              // mảng gốc
long long block[MAXN];   // tổng mỗi khối
int block_size;           // kích thước khối
int n;

void build() {
    block_size = (int)sqrt(n);  // Kích thước = √n

    // Tính tổng từng khối
    for (int i = 0; i < n; i++) {
        block[i / block_size] += a[i];
    }
}
⚠️ Lưu ý
Khối cuối cùng có thể không đủ √n phần tử. Ví dụ: n = 10, block_size = 3 → khối cuối chỉ có 1 phần tử. Không sao cả, code vẫn đúng!

Code C++

C++ Query 
long long query(int l, int r) {
    long long sum = 0;
    int bl = l / block_size;   // khối chứa l
    int br = r / block_size;   // khối chứa r

    if (bl == br) {
        // TRƯỜNG HỢP 1: l và r cùng 1 khối → duyệt trâu
        for (int i = l; i <= r; i++)
            sum += a[i];
    } else {
        // PHẦN 1: Phần lẻ bên TRÁI
        // Từ l → cuối khối bl
        for (int i = l; i < (bl + 1) * block_size; i++)
            sum += a[i];

        // PHẦN 2: Các khối NGUYÊN ở giữa
        for (int i = bl + 1; i < br; i++)
            sum += block[i];

        // PHẦN 3: Phần lẻ bên PHẢI
        // Từ đầu khối br → r
        for (int i = br * block_size; i <= r; i++)
            sum += a[i];
    }

    return sum;
}

✏️ Phần 5: Cập nhật

Cập nhật điểm: a[idx] = val

C++ Update 
void update(int idx, int val) {
    int block_id = idx / block_size;  // tìm khối
    block[block_id] -= a[idx];        // trừ giá trị cũ
    a[idx] = val;                     // gán giá trị mới
    block[block_id] += a[idx];        // cộng giá trị mới
}
// Độ phức tạp: O(1) !

Cập nhật đoạn: a[l..r] += val (với Lazy)

C++ Range Update 
long long lazy[MAXN];  // lazy cho cả khối

void update_range(int l, int r, int val) {
    int bl = l / block_size, br = r / block_size;

    if (bl == br) {
        // Cùng 1 khối → cập nhật từng phần tử
        for (int i = l; i <= r; i++) {
            a[i] += val;
            block[bl] += val;
        }
    } else {
        // Phần lẻ trái
        for (int i = l; i < (bl + 1) * block_size; i++) {
            a[i] += val;
            block[bl] += val;
        }
        // Khối nguyên → dùng lazy (không cập nhật từng phần tử)
        for (int i = bl + 1; i < br; i++)
            lazy[i] += val;
        // Phần lẻ phải
        for (int i = br * block_size; i <= r; i++) {
            a[i] += val;
            block[br] += val;
        }
    }
}

// Lấy giá trị thực tại vị trí idx
int get_val(int idx) {
    return a[idx] + lazy[idx / block_size];
}

📝 Phần 6: Code hoàn chỉnh

Tổng đoạn + Cập nhật điểm

C++ Complete 
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 100005;
int a[MAXN];
long long block[MAXN];
int block_size, n, q;

// Xây dựng
void build() {
    block_size = max(1, (int)sqrt(n));
    for (int i = 0; i < n; i++)
        block[i / block_size] += a[i];
}

// Cập nhật a[idx] = val
void update(int idx, int val) {
    int bid = idx / block_size;
    block[bid] -= a[idx];
    a[idx] = val;
    block[bid] += a[idx];
}

// Truy vấn tổng a[l..r]
long long query(int l, int r) {
    long long sum = 0;
    int bl = l / block_size, br = r / block_size;

    if (bl == br) {
        for (int i = l; i <= r; i++) sum += a[i];
    } else {
        for (int i = l; i < (bl+1) * block_size; i++) sum += a[i];
        for (int i = bl+1; i < br; i++) sum += block[i];
        for (int i = br * block_size; i <= r; i++) sum += a[i];
    }
    return sum;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    cin >> n >> q;
    for (int i = 0; i < n; i++) cin >> a[i];
    build();

    while (q--) {
        int type; cin >> type;
        if (type == 1) {
            int idx, val; cin >> idx >> val;
            update(idx, val);
        } else {
            int l, r; cin >> l >> r;
            cout << query(l, r) << "\n";
        }
    }
    return 0;
}

Cập nhật đoạn + Truy vấn điểm

C++ Range Update 
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 100005;
int a[MAXN];
long long block_sum[MAXN], lazy[MAXN];
int block_size, n, q;

void build() {
    block_size = max(1, (int)sqrt(n));
    for (int i = 0; i < n; i++)
        block_sum[i / block_size] += a[i];
}

// Cộng val vào a[l..r]
void update_range(int l, int r, int val) {
    int bl = l / block_size, br = r / block_size;
    if (bl == br) {
        for (int i = l; i <= r; i++) {
            a[i] += val;
            block_sum[bl] += val;
        }
    } else {
        for (int i = l; i < (bl+1)*block_size; i++) {
            a[i] += val; block_sum[bl] += val;
        }
        for (int i = bl+1; i < br; i++)
            lazy[i] += val;
        for (int i = br*block_size; i <= r; i++) {
            a[i] += val; block_sum[br] += val;
        }
    }
}

// Lấy giá trị thực tại idx
long long get_val(int idx) {
    return a[idx] + lazy[idx / block_size];
}

int main() {
    cin >> n >> q;
    for (int i = 0; i < n; i++) cin >> a[i];
    build();
    while (q--) {
        int type; cin >> type;
        if (type == 1) {
            int l, r, val; cin >> l >> r >> val;
            update_range(l, r, val);
        } else {
            int idx; cin >> idx;
            cout << get_val(idx) << "\n";
        }
    }
}

Tìm MIN đoạn + Cập nhật điểm

C++ Min Query 
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 100005;
int a[MAXN], block_min[MAXN];
int block_size, n, q;

void build() {
    block_size = max(1, (int)sqrt(n));
    fill(block_min, block_min + MAXN, INT_MAX);
    for (int i = 0; i < n; i++)
        block_min[i/block_size] = min(block_min[i/block_size], a[i]);
}

// Cập nhật → cần tính lại min cả khối
void update(int idx, int val) {
    a[idx] = val;
    int bid = idx / block_size;
    int start = bid * block_size;
    int end = min(start + block_size, n);
    block_min[bid] = INT_MAX;
    for (int i = start; i < end; i++)
        block_min[bid] = min(block_min[bid], a[i]);
}

// Truy vấn min a[l..r]
int query_min(int l, int r) {
    int res = INT_MAX;
    int bl = l/block_size, br = r/block_size;
    if (bl == br) {
        for (int i = l; i <= r; i++) res = min(res, a[i]);
    } else {
        for (int i = l; i < (bl+1)*block_size; i++) res = min(res, a[i]);
        for (int i = bl+1; i < br; i++) res = min(res, block_min[i]);
        for (int i = br*block_size; i <= r; i++) res = min(res, a[i]);
    }
    return res;
}

int main() {
    cin >> n >> q;
    for (int i = 0; i < n; i++) cin >> a[i];
    build();
    while (q--) {
        int type; cin >> type;
        if (type == 1) {
            int idx, val; cin >> idx >> val;
            update(idx, val);
        } else {
            int l, r; cin >> l >> r;
            cout << query_min(l, r) << "\n";
        }
    }
}

🚀 Phần 8: Nâng cao — Mo's Algorithm

Mo's Algorithm là gì?

Mo's Algorithm dùng chia căn để trả lời nhiều truy vấn offline trên đoạn một cách hiệu quả.

💡 Ý tưởng

Sắp xếp các truy vấn (l, r) theo khối của l, trong cùng khối thì sắp theo r.

Di chuyển 2 con trỏ l, r từ truy vấn này sang truy vấn kế → Tổng di chuyển O((n + q)√n).

Code Mo's Algorithm

C++ Mo's 
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 200005;
int a[MAXN], cnt[MAXN], ans[MAXN];
int block_size, n, q;
int cur_ans = 0;  // ví dụ: đếm số phần tử distinct

struct Query {
    int l, r, idx;
};

bool cmp(Query &a, Query &b) {
    int ba = a.l / block_size;
    int bb = b.l / block_size;
    if (ba != bb) return ba < bb;
    return (ba & 1) ? (a.r > b.r) : (a.r < b.r);
}

void add(int idx) {
    cnt[a[idx]]++;
    if (cnt[a[idx]] == 1) cur_ans++;  // phần tử mới xuất hiện
}

void remove(int idx) {
    cnt[a[idx]]--;
    if (cnt[a[idx]] == 0) cur_ans--;  // phần tử biến mất
}

int main() {
    cin >> n >> q;
    block_size = max(1, (int)sqrt(n));

    for (int i = 0; i < n; i++) cin >> a[i];

    vector<Query> queries(q);
    for (int i = 0; i < q; i++) {
        cin >> queries[i].l >> queries[i].r;
        queries[i].idx = i;
    }

    sort(queries.begin(), queries.end(), cmp);

    int curL = 0, curR = -1;
    for (auto &q : queries) {
        while (curR < q.r) add(++curR);
        while (curL > q.l) add(--curL);
        while (curR > q.r) remove(curR--);
        while (curL < q.l) remove(curL++);
        ans[q.idx] = cur_ans;
    }

    for (int i = 0; i < queries.size(); i++)
        cout << ans[i] << "\n";
}
✅ Độ phức tạp Mo's Algorithm

O((n + q) × √n) — rất hiệu quả cho bài toán offline với n, q ≤ 2×10⁵
